% --- Fonts (LuaLaTeX) ---
\usepackage{fontspec}
\defaultfontfeatures{Ligatures=TeX}
% Latin Modern (default) lacks Cyrillic. Set up a luaotfload font fallback so
% bare Cyrillic codepoints render via DejaVu Serif, which has a similar
% academic feel and good Cyrillic coverage. Any character not in Latin Modern
% (Cyrillic in particular) is drawn from the fallback family.
\directlua{
  luaotfload.add_fallback("cyrfallback", { "DejaVu Serif:script=cyrl;" })
}
\setmainfont{Latin Modern Roman}[RawFeature={fallback=cyrfallback}]
% Latin Modern Mono lacks box-drawing characters used in chapter 7's tree
% diagram; DejaVu Sans Mono covers the BOX DRAWING block.
\setmonofont{DejaVu Sans Mono}[Scale=MatchLowercase]
% \cyr{...} from chapter 1 is now redundant; provide a no-op so the chapter
% still compiles standalone or in the assembled document.
\providecommand{\cyr}[1]{#1}
